function generateHashtag (str) {
  if (str === '' || str.replaceAll(' ', '') === '') {
    return false
 }

let arr = str
  .split(' ')
  .filter(item => item !== '')
  .map(item => item[0].toUpperCase() + item.slice(1))
  .reduce((acc, curr) => acc + curr, '');

if (arr.length >= 140) {
  return false
}

return '#' + arr
};
